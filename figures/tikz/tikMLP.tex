\foreach \l [count=\i] in {1,2,3,n}
  \node [] at (h4-\i) {$H_{m,\l}$};

\foreach \l [count=\i] in {1,2,n}
  \draw [->] (output-\i) -- ++(1,0)
    node [above, midway] {$O_\l$};

\foreach \i in {1,...,3}
  \foreach \j in {1,...,4}
    \draw [->] (input-\i) -- (h1-\j);

  \foreach \j in {1,...,4}
